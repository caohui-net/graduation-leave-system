## P0-1修正：wrapper路径定位

**Codex指出：**
- 我的方案用`Path.home() / '.omc' / 'collaboration' / 'scripts'`
- 但项目实际路径是`<repo>/.omc/collaboration/scripts/`

```python
# .claude/skills/claude-codex-collab/scripts/collab_event.py (wrapper)
"""Wrapper for project .omc/collaboration/scripts/collab_event.py"""

import sys
from pathlib import Path
import importlib.util

def find_repo_root(start_path):
    """从当前文件向上查找repo root（包含.omc目录）"""
    current = Path(start_path).resolve()
    
    # 从.claude/skills/claude-codex-collab/scripts/向上找
    # 预期结构：<repo>/.claude/skills/claude-codex-collab/scripts/collab_event.py
    for parent in [current.parent.parent.parent.parent, current.parent.parent.parent]:
        omc_dir = parent / '.omc' / 'collaboration' / 'scripts'
        if omc_dir.exists():
            return parent
    
    raise FileNotFoundError(
        f"Cannot locate repo root with .omc/collaboration/scripts/ from {start_path}"
    )

def load_core_module():
    """加载项目内核心模块"""
    repo_root = find_repo_root(__file__)
    core_script = repo_root / '.omc' / 'collaboration' / 'scripts' / 'collab_event.py'
    
    if not core_script.exists():
        raise FileNotFoundError(
            f"Core collab_event.py not found at {core_script}. "
            f"Run collab_init.py first."
        )
    
    # 临时添加scripts目录到sys.path（支持sibling imports）
    scripts_dir = str(core_script.parent)
    original_path = sys.path.copy()
    
    try:
        if scripts_dir not in sys.path:
            sys.path.insert(0, scripts_dir)
        
        # 用唯一名称加载，避免与wrapper自身冲突
        spec = importlib.util.spec_from_file_location(
            "_omc_collab_event_core", 
            core_script
        )
        module = importlib.util.module_from_spec(spec)
        spec.loader.exec_module(module)
        
        return module
    finally:
        # 恢复sys.path
        sys.path[:] = original_path

# 加载核心模块
_core = load_core_module()

# Re-export API
append_event = _core.append_event
acquire_lock = _core.acquire_lock
release_lock = _core.release_lock

# CLI入口
if __name__ == '__main__':
    _core.main()
```

2. ✅ 验证`.omc/collaboration/scripts/`存在
3. ✅ 临时添加scripts目录到sys.path（支持sibling imports）
4. ✅ try/finally恢复sys.path
5. ✅ 用唯一模块名避免命名冲突

**验证点：**

## P0-3修正：claim_task()原子校验

**Codex指出：**
- 简化为`append_event("task_claimed")`会丢失claim语义
- claim需要：持锁检查active owner、重建任务生命周期、确认可认领
- 普通append_event()没有这些领域校验

**Claude修正方案：**

**方案：在核心模块提供专门的claim_task() API**

```python
# .omc/collaboration/scripts/collab_task.py (新核心模块)

from pathlib import Path
import json
from collab_lock import acquire_lock, release_lock
from collab_journal import get_max_event_id, parse_events
from collab_event import append_event_locked

def claim_task(task_id, agent_name):
    """
    原子认领任务。
    
    持锁期间：
    1. 检查任务状态（是否已被认领、是否已完成）
    2. 验证agent可以认领
    3. 追加task_claimed事件
    4. 更新state
    
    失败场景：
    - 任务已被其他agent认领且仍active
    - 任务已完成/取消
    - 任务不存在
    """
    LOCK_DIR = Path('.omc/collaboration/locks/journal.lock')
    STATE_FILE = Path('.omc/collaboration/state.json')
    EVENTS_FILE = Path('.omc/collaboration/events.jsonl')
    
    lock = acquire_lock(LOCK_DIR, agent_name)
    
    try:
        # 1. 读取当前state
        with open(STATE_FILE, 'r') as f:
            current_state = json.load(f)
        
        # 2. 重建任务生命周期（从events.jsonl）
        events = parse_events(EVENTS_FILE, strict=True)
        task_events = [e for e in events if e.get('details', {}).get('task_id') == task_id]
        
        if not task_events:
            raise ValueError(f"Task {task_id} not found in event log")
        
        # 3. 检查任务当前状态
        task_status = None
        task_owner = None
        for event in task_events:
            if event['type'] == 'task_created':
                task_status = 'open'
            elif event['type'] == 'task_claimed':
                task_status = 'claimed'
                task_owner = event.get('agent')
            elif event['type'] == 'task_completed':
                task_status = 'completed'
            elif event['type'] == 'task_cancelled':
                task_status = 'cancelled'
        
        # 4. 验证可认领
        if task_status == 'completed':
            raise ValueError(f"Task {task_id} already completed")
        if task_status == 'cancelled':
            raise ValueError(f"Task {task_id} already cancelled")
        if task_status == 'claimed' and task_owner != agent_name:
            raise ValueError(
                f"Task {task_id} already claimed by {task_owner}"
            )
        
        # 5. 追加task_claimed事件（使用locked版本，避免重新获取锁）
        event_id = get_max_event_id(EVENTS_FILE) + 1
        
        event = {
            'id': event_id,
            'type': 'task_claimed',
            'agent': agent_name,
            'timestamp': datetime.utcnow().isoformat() + 'Z',
            'details': {
                'task_id': task_id
            }
        }
        
        # 追加到events.jsonl
        with open(EVENTS_FILE, 'a') as f:
            f.write(json.dumps(event) + '\n')
            f.flush()
        
        # 6. 更新state
        current_state['current_task'] = task_id
        current_state['active_agent'] = agent_name
        current_state['status'] = 'in_progress'
        current_state['last_event_id'] = event_id
        current_state['updated_at'] = event['timestamp']
        
        temp_state = STATE_FILE.with_name(f'state.json.tmp.{agent_name}')
        with open(temp_state, 'w') as f:
            json.dump(current_state, f, indent=2)
        
        # 验证temp state
        with open(temp_state, 'r') as f:
            verified = json.load(f)
        if not isinstance(verified, dict):
            raise ValueError("temp state is not a JSON object")
        
        # 原子rename
        temp_state.rename(STATE_FILE)
        
    finally:
        release_lock(lock, agent_name)
```

**关键修正：**
1. ✅ 持锁期间完成所有操作
2. ✅ 从events.jsonl重建任务生命周期
3. ✅ 验证任务状态（open/claimed/completed/cancelled）
4. ✅ 检查是否已被其他agent认领
5. ✅ 在同一个锁内写事件和state
6. ✅ 避免claim竞态

## P0-4修正：release_lock() owner字段

**Codex指出：**
- 我的方案读取`owner_data.get('owner')`
- 但协议和acquire_lock()写的是`agent`字段
- owner匹配永远失败，正常路径会留下锁

**Claude修正方案：**

```python
# .omc/collaboration/scripts/collab_lock.py

def acquire_lock(lock_dir, agent_name, task_id=None):
    """
    获取锁。使用mkdir原子性。
    
    写入owner.json格式：
    {
      "agent": "claude",
      "task_id": "TASK-001",  // 可选
      "acquired_at": "2026-05-30T17:00:00.000Z"
    }
    """
    from datetime import datetime
    import json
    from pathlib import Path
    
    lock_dir = Path(lock_dir)
    
    try:
        lock_dir.mkdir(parents=True, exist_ok=False)
    except FileExistsError:
        # 锁已存在，检查是否stale
        owner_file = lock_dir / 'owner.json'
        if owner_file.exists():
            try:
                with open(owner_file, 'r') as f:
                    owner_data = json.load(f)
                acquired_at = owner_data.get('acquired_at')
                # 报告stale lock，但不删除
                print(f"⚠️  Lock held by {owner_data.get('agent')} since {acquired_at}")
            except:
                print(f"⚠️  Lock exists with malformed owner.json")
        
        raise ValueError(f"Lock {lock_dir} already held")
    
    # 写入owner.json
    owner_data = {
        'agent': agent_name,
        'acquired_at': datetime.utcnow().isoformat() + 'Z'
    }
    if task_id:
        owner_data['task_id'] = task_id
    
    owner_file = lock_dir / 'owner.json'
    with open(owner_file, 'w') as f:
        json.dump(owner_data, f, indent=2)
    
    return lock_dir

def release_lock(lock_dir, agent_name, task_id=None):
    """
    释放锁。只在agent匹配时删除。malformed owner.json → 失败。
    
    参数：
    - lock_dir: 锁目录
    - agent_name: 期望的agent名称
    - task_id: 可选的task_id校验
    """
    from pathlib import Path
    import json
    
    lock_dir = Path(lock_dir)
    owner_file = lock_dir / 'owner.json'
    
    if not owner_file.exists():
        raise ValueError(
            f"Lock {lock_dir} has no owner.json - cannot release"
        )
    
    try:
        with open(owner_file, 'r') as f:
            owner_data = json.load(f)
    except (json.JSONDecodeError, IOError) as e:
        # malformed owner.json → 失败，不删除锁
        raise ValueError(
            f"Lock {lock_dir} has malformed owner.json: {e}. "
            f"Cannot release. Use collab_validate.py repair."
        )
    
    # 校验agent字段（不是owner字段）
    stored_agent = owner_data.get('agent')
    if stored_agent != agent_name:
        raise ValueError(
            f"Lock {lock_dir} held by agent={stored_agent!r}, "
            f"cannot release by agent={agent_name!r}"
        )
    
    # 可选：校验task_id
    if task_id:
        stored_task = owner_data.get('task_id')
        if stored_task and stored_task != task_id:
            raise ValueError(
                f"Lock {lock_dir} held for task={stored_task!r}, "
                f"cannot release for task={task_id!r}"
            )
    
    # agent匹配，删除锁
    try:
        owner_file.unlink()
        lock_dir.rmdir()
    except OSError as e:
        raise ValueError(f"Failed to remove lock {lock_dir}: {e}")
```

**关键修正：**
1. ✅ acquire_lock()写入`agent`字段（不是owner）
2. ✅ release_lock()读取`agent`字段（不是owner）
3. ✅ 字段命名与协议一致
4. ✅ 可选task_id校验
5. ✅ malformed owner.json → 失败，不删除

**验证点：**
- release_lock()覆盖：matching agent、mismatched agent、missing owner.json、malformed owner.json
- 正常写入路径不会留下锁
- malformed owner需要显式repair
